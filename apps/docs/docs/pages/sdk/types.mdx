# Types

Key TypeScript types exported from `@0xslots/sdk`.

[Source](https://github.com/nezz0746/0xSlots/blob/main/packages/sdk/src/client.ts)

## `SlotsChain`

Supported chain IDs.

```ts
enum SlotsChain {
  BASE = 8453,
  BASE_SEPOLIA = 84532,
  ARBITRUM = 42161,
}
```

## `SlotsClientConfig`

Configuration for creating a `SlotsClient`.

```ts
interface SlotsClientConfig {
  chainId: SlotsChain;
  factoryAddress?: Address;
  publicClient?: PublicClient;
  walletClient?: WalletClient;
  subgraphUrl?: string;
  subgraphApiKey?: string;
  headers?: Record<string, string>;
}
```

## `SlotConfig`

Immutable configuration for a slot, set at deployment.

```ts
interface SlotConfig {
  mutableTax: boolean;
  mutableModule: boolean;
  manager: Address;
}
```

## `SlotInitParams`

Initial parameters for slot deployment.

```ts
interface SlotInitParams {
  taxPercentage: bigint;          // Basis points per month (100 = 1%)
  module: Address;                 // Hook contract (or zero address)
  liquidationBountyBps: bigint;   // Bounty for liquidators (basis points)
  minDepositSeconds: bigint;      // Minimum deposit in seconds of tax coverage
}
```

## `CreateSlotParams`

Parameters for `createSlot()`.

```ts
interface CreateSlotParams {
  recipient: Address;
  currency: Address;
  config: SlotConfig;
  initParams: SlotInitParams;
}
```

## `CreateSlotsParams`

Parameters for `createSlots()` (batch deployment).

```ts
interface CreateSlotsParams extends CreateSlotParams {
  count: bigint;
}
```

## `BuyParams`

Parameters for `buy()`.

```ts
interface BuyParams {
  slot: Address;
  depositAmount: bigint;
  selfAssessedPrice: bigint;
}
```

## `SlotOnChain`

On-chain slot state returned by `useSlotOnChain` and `useSlotsOnChain` hooks.

```ts
type SlotOnChain = {
  // Identity
  id: string;
  recipient: string;
  currency: string;
  manager: string;
  mutableTax: boolean;
  mutableModule: boolean;

  // State
  occupant: string | null;         // null when vacant
  price: bigint;
  taxPercentage: bigint;
  module: string;
  liquidationBountyBps: bigint;
  minDepositSeconds: bigint;

  // Financials
  deposit: bigint;
  collectedTax: bigint;
  taxOwed: bigint;
  secondsUntilLiquidation: bigint;
  insolvent: boolean;

  // Pending updates
  hasPendingTax: boolean;
  pendingTaxPercentage: bigint;
  hasPendingModule: boolean;
  pendingModule: string;

  // Currency metadata (from ERC-20)
  currencyName?: string;
  currencySymbol?: string;
  currencyDecimals?: number;
};
```

## `SlotActionCallbacks`

Callback interface for `useSlotAction`.

```ts
interface SlotActionCallbacks {
  onSuccess?: (label: string, hash: Hash) => void;
  onError?: (label: string, error: string) => void;
}
```
